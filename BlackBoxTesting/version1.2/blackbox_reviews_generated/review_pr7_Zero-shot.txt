def parse_user(data: dict) -> dict:
    """
    Parse user data from a dictionary.

    Args:
        data (dict): A dictionary containing 'name' and 'age' keys.

    Returns:
        dict: A dictionary with parsed user data.

    Raises:
        ValueError: If the input data is invalid.
    """
    if not isinstance(data, dict):
        raise ValueError("Input data must be a dictionary")

    required_keys = ['name', 'age']
    if not all(key in data for key in required_keys):
        raise ValueError("Input data must contain 'name' and 'age' keys")

    try:
        name = data['name'].strip()
        age = int(data['age'])
        return {'name': name, 'age': age}
    except ValueError as e:
        raise ValueError("Invalid age value") from e
