def sum_range(a: int, b: int) -> int:
    """
    Returns the inclusive sum of integers from a to b.

    Args:
        a (int): The lower bound of the range (inclusive).
        b (int): The upper bound of the range (inclusive).

    Returns:
        int: The sum of integers from a to b.
    """
    return (b - a + 1) * (a + b) // 2
