def sum_range(a: int, b: int) -> int:
    """
    Returns the inclusive sum of integers from a to b.
    
    Args:
    a (int): The start of the range (inclusive).
    b (int): The end of the range (inclusive).
    
    Returns:
    int: The sum of integers from a to b.
    """
    if a > b:
        a, b = b, a  # Swap a and b if a is greater than b
    return (b - a + 1) * (a + b) // 2
